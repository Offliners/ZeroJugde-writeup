題目:

給你N個數字X1~XN，求出X1X2+X1X3+X1X4+......+XN-2XN-1+XN-2XN+XN-1XN

## C
```C
#include<stdio.h>

int main(void)
{
	int n, temp;
	
	while(scanf("%d", &n) != EOF)
	{
		long long int sum = 0;
		long long int result = 0;
		
		for(int i = 0; i < n; i++)
		{
			scanf("%d", &temp);
			
			result -= temp * temp;
			sum += temp;
		}
		
		printf("%lld\n", (result + sum * sum) / 2);
	}
	
	return 0;
} 
```
