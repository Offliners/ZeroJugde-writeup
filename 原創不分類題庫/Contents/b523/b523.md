## C++
```C++
#include<iostream>
#include<set>

using namespace std;

int main(void)
{
	string str;
	set<string> s;
	
	while(getline(cin, str))
	{
		cout << (s.count(str) ? "YES" : "NO") << endl;
		
		s.insert(str);
	}
	
	return 0;
}
```
 * Result : `AC (0.3s, 2.3MB)`

## Python
```python
import sys
from collections import OrderedDict

known = OrderedDict()
i = 0

for temp in sys.stdin:
    temp = temp.strip()
    if  known.get(temp) != None:
        print('YES')
    else:
        print('NO')
        known[temp] = i
        i += 1
```
 * Result : `AC (29ms, 5.8MB)`
